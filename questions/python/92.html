# create a json string dictionary that has name, age, and city # convert the string into a dictionary using json loads # print a property import json # some JSON: x = '{ "name":"John", "age":30, "city":"New York"}' # parse x: y = json.loads(x) # the result is a Python dictionary: print(y["age"]) # 30 # create a dictionary that has name, age, and city # convert it into a json string using dumps import json x = { "name": "John", "age": 30, "city": "New York" } # convert into JSON: y = json.dumps(x) # the result is a JSON string: print(y) # {"name": "John", "age": 30, "city": "New York"} # create a dictionary that has name, age, and city # convert it into a json string using dumps with indent and sort keys for readability print(json.dumps(x, indent=4, sort_keys=True)) # { "age": 30, "cars": [ { "model": "BMW 230", "mpg": 27.5 }, { "model": "Ford Edge", "mpg": 24.1 } ], "children": [ "Ann", "Billy" ], "divorced": false, "married": true, "name": "John", "pets": null }
