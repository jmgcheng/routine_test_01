# combine a range of 5 and a hello string in a single list values = [*range(5), *"Hello"] # [0, 1, 2, 3, 4, 'H', 'e', 'l', 'l', 'o'] # combine 2 dictionaries first = {"x": 1} second = {"x": 10, "y": 2} combined = {**first, **second, "z": 1} print(combined) # {'x': 10, 'y': 2, 'z': 1}
